446 /*********************************************************************
447  * Private методы класса RxIoSampleResponse.
448  *********************************************************************/
449  void RxIoSampleResponse::getNextSample(uint8_t &alias)
450  {
451  switch(getData()[alias] & 0x7F) //Сбрасываем старший бит, в котором передается информация о текущем состоянии цифровой линии.
452  {
453  case IO_ADC:
454  alias += 4;
455  break;
456  case IO_DIGITAL_INPUT:
457  case IO_DIGITAL_OUTPUT_LO:
458  case IO_DIGITAL_OUTPUT_HI:
459  case IO_WAKEUP_INPUT_FALLING_EDGE:
460  case IO_WAKEUP_INPUT_RISING_EDGE:
461  alias += 2;
462  break;
463  case IO_COUNTER_INPUT1:
464  case IO_COUNTER_INPUT2:
465  alias += 6;
466  break;
467  }
468  }

470 uint8_t RxIoSampleResponse::getSample(uint8_t number)
471 {
472  uint8_t count = 1;
473  uint8_t i = 1;
474  uint8_t & iRef = i;
475  while(i < getDataLength())
476  {
477  if(count == number)
478  break;
479  getNextSample(iRef);
480  count++;
481  }
482  return i;
483 }

761 uint8_t RemoteAtCommandRequest::getFrameData(uint8_t pos)
762 {
763  if(pos == 0)
764  {
765  return (_remoteAddress >> 8) & 0xff;
766  }
767  else if(pos == 1)
768  {
769  return _remoteAddress & 0xff;
770  }
771  else if(pos == 2)
772  {
773  return ((_acknowledge ? 0: BV(ACKNOWLEDGE_DISABLE_COMMAND_OPTIONS_BIT)) +\
774  (_applyChanges ? BV(APPLY_CHANGES_COMMAND_OPTION_BIT): 0) +\
775  (_saveChanges ? 0: BV(APPLY_CHANGES_NO_SAVE_COMMAND_OPTION_BIT)) +\
776  (_cca ? 0: BV(CCA_DISABLE_COMMAND_OPTION_BIT)) +\
777  (_encryption ? BV(ENCRYPTION_ENABLE_COMMAND_OPTION_BIT): 0) +\
778  (_sleepingDevice ? BV(SLEEP_DEVICE_COMMAND_OPTIONS_BIT): 0));
779  }
780  else if(pos == 3)
781  {
782  return getCommand()[0];
783  }
784  else if (pos == 4)
785  {
786  return getCommand()[1];
787  }
788  else
789  {
790  return getCommandValue()[pos - REMOTE_AT_COMMAND_API_LENGTH];
791  }
792 }

971 /*********************************************************************
972  * Переопределенные Public методы класса SerialStar, наследуемые из MBee868
973  *********************************************************************/
974 void SerialStar::readPacket()
975 {
976  if(_response.isAvailable() || _response.isError())
977  resetResponse(); //Сброс предыдущих сообщений.
978  while(available())
979  {
980  b = read();
981  if(_pos > 0 && b == START_BYTE && ATAP == 2)
982  {
983  //Обнаружен стартовый байт до окончания обработки предыдущего пакета. Сбрасываем предыдущее сообщение и начинаем разбор заново.
984  _response.setErrorCode(UNEXPECTED_START_BYTE);
985  return;
986  }
987  if(_pos > 0 && b == ESCAPE)
988  {
989  if(available())
990  {
991  b = read();
992  b = 0x20 ^ b; //Извлекаем символ из escape-последовательности.
993  }
994  else
995  {
996  _escape = true; //Ставим флаг того, что следующий байт будет являться вторым символом из escape-последовательности.
997  continue;
998  }
999  }
1000  if(_escape == true)
1001  {
1002  b = 0x20 ^ b;
1003  _escape = false; //Предыдущий байт был escape-символом. Извлекаем символ из escape-последовательности.
1004  }
1005  if(_pos >= API_ID_INDEX) //Контрольную сумму считаем начиная с байта типа API-фрейма.
1006  _checksumTotal+= b;
1007 
1008  //Конечный автомат парсера.
1009  switch(_pos)
1010  {
1011  case 0:
1012  if(b == START_BYTE)
1013  _pos++;
1014  break;
1015  case 1:
1016  _response.setMsbLength(b);
1017  _pos++;
1018  break;
1019  case 2:
1020  _response.setLsbLength(b);
1021  _pos++;
1022  break;
1023  case 3:
1024  _response.setApiId(b);
1025  _pos++;
1026  break;
1027  default:
1028  //Далее разбираются байты, следующие за 4-ым.
1029  if(_pos > MAX_FRAME_DATA_SIZE)
1030  {
1031  _response.setErrorCode(PACKET_EXCEEDS_BYTE_ARRAY_LENGTH); //Длина пакета превышает максимально допустимую.
1032  return;
1033  }
1034  //Проверяем, не достигли ли еще конца поля данных.
1035  if(_pos == (_response.getPacketLength() + 3))
1036  {
1037  if((_checksumTotal & 0xff) == 0xff)
1038  {
1039  _response.setChecksum(b);
1040  _response.setAvailable(true);
1041  _response.setErrorCode(NO_ERROR);
1042  }
1043  else
1044  _response.setErrorCode(CHECKSUM_FAILURE); //Ошибка контрольной суммы.
1045  _response.setFrameLength(_pos - 4); //Длина поля данных не включает также поле Frame Type.
1046  _pos = 0;
1047  return;
1048  }
1049  else
1050  {
1051  _response.getFrameData()[_pos - 4] = b; //Добавляем считанный символ в поле данных.
1052  _pos++;
1053  }
1054  }
1055  }
1056 }
